**File 1: `src/lib/utils.ts`**
The cn() utility is the standard shadcn class-merging helper. It uses clsx to join class values and tailwind-merge to deduplicate conflicting Tailwind classes.

```typescript
// src/lib/utils.ts
// Standard shadcn cn() utility — used by Button, Select, Tooltip copy-paste components.
// clsx handles conditional/array class values; tailwind-merge resolves Tailwind conflicts.
import { type ClassValue, clsx } from 'clsx';
import { twMerge } from 'tailwind-merge';

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}
```

**File 2: `src/components/ExplainContext.tsx`**
React Context for the global explainMode boolean. The provider reads localStorage on mount (same pattern as DashboardHeader theme toggle). SectionHeader components will call useExplainMode() to read the state — no prop drilling needed.

IMPORTANT: This file needs 'use client' because it uses useState/useEffect. It renders inside DashboardApp's existing client boundary — no new RSC boundary is created.

```typescript
// src/components/ExplainContext.tsx
// React Context for global explainMode toggle.
// Pattern: localStorage read on mount, same as DashboardHeader theme toggle.
// Provider wraps inside Redux <Provider> in DashboardApp (Plan 10-05).
// SectionHeader calls useExplainMode() directly — no prop threading needed.
'use client';
import { createContext, useContext, useState, useEffect, type ReactNode } from 'react';

interface ExplainContextValue {
  explainMode: boolean;
  toggleExplainMode: () => void;
}

const ExplainContext = createContext<ExplainContextValue>({
  explainMode: false,
  toggleExplainMode: () => {},
});

export function ExplainProvider({ children }: { children: ReactNode }) {
  const [explainMode, setExplainMode] = useState(false);

  // Initialize from localStorage on mount — persists across page refreshes.
  // Silent catch: localStorage may be blocked in some environments (e.g., private mode).
  useEffect(() => {
    try {
      const stored = localStorage.getItem('explainMode');
      if (stored === 'true') setExplainMode(true);
    } catch (_) {}
  }, []);

  const toggleExplainMode = () => {
    setExplainMode(prev => {
      const next = !prev;
      try { localStorage.setItem('explainMode', String(next)); } catch (_) {}
      return next;
    });
  };

  return (
    <ExplainContext.Provider value={{ explainMode, toggleExplainMode }}>
      {children}
    </ExplainContext.Provider>
  );
}

// Hook used by SectionHeader and DashboardHeader to read/write explain mode.
export function useExplainMode(): ExplainContextValue {
  return useContext(ExplainContext);
}
```
